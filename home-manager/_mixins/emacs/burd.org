* Configuration
  Emacs is a special beast. Taming it takes a lot of care. In an
  attempt to document/explain/share with the rest of the world, this
  is my attempt at configuration as a literate program. It also shows
  off the awesome power of org-mode, which makes all of this possible.

  This configuration has been modernized to use:
  - use-package for declarative package management
  - vertico/consult/orderless for modern completion
  - company for auto-completion
  - eglot for LSP support (with comprehensive language servers)
  - electric-pair-mode for bracket pairing
  - lexical binding throughout
  - tsdh-dark theme (built-in, good contrast)

*** Font Ligatures
    :PROPERTIES:
    :CUSTOM_ID: ligature
    :END:

    Enable Fira Code ligatures for programming symbols.
    #+begin_src emacs-lisp
      (defun fira-code-mode--make-alist (list)
        "Generate prettify-symbols alist from LIST."
        (let ((idx -1))
          (mapcar
           (lambda (s)
             (setq idx (1+ idx))
             (let* ((code (+ #Xe100 idx))
                    (width (string-width s))
                    (prefix ())
                    (suffix '(?\s (Br . Br)))
                    (n 1))
               (while (< n width)
                 (setq prefix (append prefix '(?\s (Br . Bl))))
                 (setq n (1+ n)))
               (cons s (append prefix suffix (list (decode-char 'ucs code))))))
           list)))

      (defconst fira-code-mode--ligatures
        '("www" "**" "***" "**/" "*>" "*/" "\\\\" "\\\\\\"
          "{-" "[]" "::" ":::" ":=" "!!" "!=" "!==" "-}"
          "--" "---" "-->" "->" "->>" "-<" "-<<" "-~"
          "#{" "#[" "##" "###" "####" "#(" "#?" "#_" "#_("
          ".-" ".=" ".." "..<" "..." "?=" "??" ";;" "/*"
          "/**" "/=" "/==" "/>" "//" "///" "&&" "||" "||="
          "|=" "|>" "^=" "$>" "++" "+++" "+>" "=:=" "=="
          "===" "==>" "=>" "=>>" "<=" "=<<" "=/=" ">-" ">="
          ">=>" ">>" ">>-" ">>=" ">>>" "<*" "<*>" "<|" "<|>"
          "<$" "<$>" "<!--" "<-" "<--" "<->" "<+" "<+>" "<="
          "<==" "<=>" "<=<" "<>" "<<" "<<-" "<<=" "<<<" "<~"
          "<~~" "</" "</>" "~@" "~-" "~=" "~>" "~~" "~~>" "%%"
          "x" ":" "+" "+" "*"))

      (defvar fira-code-mode--old-prettify-alist)

      (defun fira-code-mode--enable ()
        "Enable Fira Code ligatures in current buffer."
        (setq-local fira-code-mode--old-prettify-alist prettify-symbols-alist)
        (setq-local prettify-symbols-alist
                    (append (fira-code-mode--make-alist fira-code-mode--ligatures)
                            fira-code-mode--old-prettify-alist))
        (prettify-symbols-mode t))

      (defun fira-code-mode--disable ()
        "Disable Fira Code ligatures in current buffer."
        (setq-local prettify-symbols-alist fira-code-mode--old-prettify-alist)
        (prettify-symbols-mode -1))

      (define-minor-mode fira-code-mode
        "Fira Code ligatures minor mode"
        :lighter " Fira Code"
        (setq-local prettify-symbols-unprettify-at-point 'right-edge)
        (if fira-code-mode
            (fira-code-mode--enable)
          (fira-code-mode--disable)))

      (defun fira-code-mode--setup ()
        "Setup Fira Code Symbols"
        (set-fontset-font t '(#Xe100 . #Xe16f) "Fira Code Symbol"))

      (provide 'fira-code-mode)
    #+end_src

        "Comments or uncomments the region or the current line if there's no active region."
        (interactive)
        (let (beg end)
          (if (region-active-p)
              (setq beg (region-beginning) end (region-end))
            (setq beg (line-beginning-position) end (line-end-position)))
          (comment-or-uncomment-region beg end)

*** LSP Support (eglot)
    :PROPERTIES:
    :CUSTOM_ID: eglot
    :END:

    Modern LSP client with comprehensive language server support.

    Useful keybindings (in buffers with eglot active):
    - =M-.= Jump to definition
    - =M-,= Return to previous location
    - =M-?= Find references
    - =C-h .= Show documentation at point
    - =C-c l r= Rename symbol
    - =C-c l f= Format buffer/region
    - =C-c l a= Code actions

    #+begin_src emacs-lisp
      (use-package eglot
        :ensure nil  ;; Built-in in Emacs 29+
        :hook ((c-mode . eglot-ensure)
               (c++-mode . eglot-ensure)
               (rust-mode . eglot-ensure)
               (go-mode . eglot-ensure)
               (python-mode . eglot-ensure)
               (sh-mode . eglot-ensure)
               (bash-ts-mode . eglot-ensure))
        :bind (:map eglot-mode-map
                    ("C-c l r" . eglot-rename)
                    ("C-c l f" . eglot-format)
                    ("C-c l a" . eglot-code-actions))
        :custom
        ;; Increase timeout for slow language servers
        (eglot-connect-timeout 60)
        ;; Shutdown server when last buffer is closed
        (eglot-autoshutdown t)
        ;; Send more complete information to the server
        (eglot-send-changes-idle-time 0.5)
        :config
        ;; Ensure clangd from Homebrew LLVM is used (better than system)
        (when (file-exists-p "/opt/homebrew/opt/llvm/bin/clangd")
          (add-to-list 'eglot-server-programs
                       '((c-mode c++-mode) . ("/opt/homebrew/opt/llvm/bin/clangd"
                                               "--background-index"
                                               "--clang-tidy"
                                               "--completion-style=detailed"
                                               "--header-insertion=never"))))

        ;; Bash Language Server
        (add-to-list 'eglot-server-programs
                     '((sh-mode bash-ts-mode) . ("bash-language-server" "start")))

        ;; SQL Language Server (for PostgreSQL)
        (add-to-list 'eglot-server-programs
                     '(sql-mode . ("sql-language-server" "up" "--method" "stdio")))

        ;; Rust analyzer with better settings
        (add-to-list 'eglot-server-programs
                     '(rust-mode . ("rust-analyzer")))

        ;; Python with pyright
        (add-to-list 'eglot-server-programs
                     '(python-mode . ("pyright-langserver" "--stdio")))

        ;; Go with gopls

      (defun get-closest-pathname (&optional max-level file)
        "Find FILE by searching up directory tree up to MAX-LEVEL levels."
        (let ((root (expand-file-name "/"))
              (max-level (or max-level 3))
              (file (or file "Makefile"))
              (level 0))
          (expand-file-name file
                            (cl-loop
                             for d = default-directory then (expand-file-name ".." d)
                             do (setq level (+ level 1))
                             if (file-exists-p (expand-file-name file d))
                             return d
                             if (> level max-level)
                             return nil
                             if (equal d root)
                             return nil))))

      (add-hook 'c-mode-hook
                (lambda ()
                  (unless (file-exists-p "Makefile")
                    (setq-local compile-command
                                (let ((file (file-name-nondirectory buffer-file-name))
                                      (mkfile (get-closest-pathname)))
                                  (if mkfile
                                      (format "cd %s; make -f %s"
                                              (file-name-directory mkfile) mkfile)
                                    (format "%s -c -o %s.o %s %s %s"
                                            (or (getenv "CC") "gcc")
                                            (file-name-sans-extension file)
                                            (or (getenv "CPPFLAGS") "-DDEBUG=9")
                                            (or (getenv "CFLAGS") "-ansi -pedantic -Wall -g")
                                            file)))))))

      (defvar gdb-libtool-command-name "libtool"
        "Pathname for executing gdb through libtool.")

      (defun gdb-libtool (path &optional corefile)
        "Run gdb on a libtool program FILE in buffer *gdb-FILE*.
      PATH is the path to the program.
      Optional COREFILE is the core file to debug."
        (interactive "FRun gdb-libtool on file: ")
        (load "gud")
        (setq path (file-truename (expand-file-name path)))
        (let ((file (file-name-nondirectory path)))
          (switch-to-buffer (concat "*gud-" file "*"))
          (setq default-directory (file-name-directory path))
          (or (bolp) (newline))
          (insert "Current directory is " default-directory "\n")
          (apply 'make-comint
                 (concat "gud-" file)
                 (substitute-in-file-name gdb-libtool-command-name)
                 nil
                 "--mode=execute"
                 (substitute-in-file-name gdb-command-name)
                 "-fullname"
                 "-cd" default-directory
                 file
                 (and corefile (list corefile)))
          (gud-mode)
          (gud-set-buffer)))
